# Mercedes-Benz Automotive AI - Production Dockerfile
FROM nvidia/cuda:11.8-devel-ubuntu22.04

# Set working directory
WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip \
    python3-dev \
    git \
    wget \
    curl \
    ffmpeg \
    libsm6 \
    libxext6 \
    libfontconfig1 \
    libxrender1 \
    libgl1-mesa-glx \
    libasound2-dev \
    portaudio19-dev \
    && rm -rf /var/lib/apt/lists/*

# Upgrade pip
RUN python3 -m pip install --upgrade pip

# Copy requirements first for better Docker layer caching
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY . .

# Create necessary directories
RUN mkdir -p data/models data/logs data/cache

# Set Python path
ENV PYTHONPATH=/app

# Create non-root user for security
RUN useradd -m mercedes_ai && chown -R mercedes_ai:mercedes_ai /app
USER mercedes_ai

# Expose port for API
EXPOSE 8000

# Health check
HEALTHCHECK --interval=30s --timeout=30s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:8000/health || exit 1

# Command to run the application
CMD ["python3", "main.py"]

# Alternative commands for different deployment modes:
# For API server: CMD ["uvicorn", "src.api:app", "--host", "0.0.0.0", "--port", "8000"]
# For training: CMD ["python3", "scripts/train_model.py"]
# For batch processing: CMD ["python3", "scripts/batch_process.py"]
